import Common from "./Common";
export default {
  name: "FSwitch",
  mixins: [Common],
  computed: {
    activeValue() {
      if (!this.data.enumJson.length) return "1";
      const item = this.data.enumJson[0];
      if (!item) return "1";
      return item.value;
    },
    inActiveValue() {
      if (!this.data.enumJson.length) return "0";
      const item = this.data.enumJson[1];
      if (!item) return "0";
      return item.value;
    }
  }
};
